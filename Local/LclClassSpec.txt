Note: All terms not defined here should be referenced in AutoFunctorSpec.txt

memlike:
  1) When implicitly typed, the types of the declared member variables are 
    deduced from variables of the same name in the enclosing scope(s) using
    Boost.TypeOf
  2) The signature of the generated constructor for the local class will reflect 
    what's specified here.

memset:
  1) Specifies the member variables of the local class that will be initialized 
    to some value during construction.

//---
//BNF
//---

special symbols:
  [...] specifies an optional BNF expression.
terminals: derives memlike memset

lclclass-usage :=
  lclclass-start-macro ( name , lclclass-spec )
  member-specification
  WG_LCLCLASS_END [;]
  
lclclass-start-macro :=
    WG_LCLCLASS
  | WG_LCLCLASS_TPL

lclclass-spec :=
  [derives derives-seq]
  [memlike memlike-tuple-seq]
  [memset memset-seq]

derives-seq := derives-tuple-seq

derives-tuple-seq :=
    derives-tuple
  | derives-tuple-seq derives-tuple

derives-tuple :=
  ( base-specifier )

memlike-tuple-seq :=
    memlike-tuple
  | memlike-tuple-seq memlike-tuple

memlike-tuple :=
  ( type-var-dcln )

memset-seq := set-tuple-seq

member-specification := See C++ standard.
base-specifier := See C++ standard.
